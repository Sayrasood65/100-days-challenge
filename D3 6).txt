6) Write a program to swap two numbers using a third variable.

#include <stdio.h>

int main() {
    int a , b , swap ;
    printf("enter first num:");
    scanf("%d", &a);
     printf("enter second num:");
     scanf("%d", &b);
     
     printf("before swapping:");
     printf("%d %d\n", a , b);
     
     swap = a;
     a = b;
     b = swap;
    printf("after swapping:");
    printf("%d %d\n", a , b);
    
    return 0;
}
